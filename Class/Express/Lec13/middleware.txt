Middle ware is a function that execute during request before sending the final response. They have access through
3 parameters like request, response and next.
client -> middleware -> server

Syntax :




const express = require('express')
const app = express()
const port = 3000

function checkRoute(req, res, next) {
    console.log(req.url);
    
}

app.use(checkRoute);


app.get('/', (req, res) => {
  res.send('Hello World!')
})

app.listen(port, () => {
  console.log(`Example app listening on port ${port}`)
})
